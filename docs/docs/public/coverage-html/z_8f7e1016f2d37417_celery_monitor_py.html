25 def monitor_worker(): 

26 retry_interval = 5 # Start with 5 seconds

33 def on_task_started(event):

43 def on_task_succeeded(event): 

44 task_info = self.active_tasks.pop(event['uuid'], {}) 

45 logger.info(f"✅ Task completed: {task_info.get('name', 'unknown')} ({event['uuid'][:8]})")

47 def on_task_failed(event): 

48 task_info = self.active_tasks.pop(event['uuid'], {}) 

49 logger.warning(f"❌ Task failed: {task_info.get('name', 'unknown')} ({event['uuid'][:8]})")

51 def on_worker_heartbeat(event): 

52 self.worker_stats[event['hostname']] = { 

53 'last_heartbeat': datetime.fromtimestamp(event['timestamp']).isoformat(), 

54 'status': 'online' 

55 } 

56 

57 # Attempt connection 

58 with celery_app.connection() as connection: 

59 receiver = EventReceiver( 

60 connection, 

61 handlers={

64 'task-failed': on_task_failed, 

65 'worker-heartbeat': on_worker_heartbeat,
